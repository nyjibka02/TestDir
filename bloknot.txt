ls - список папоr, файлов
cd - переход в директорию
mkdir - создание директории
touch - создание файлов
cp - копировать
mv - переименовать
echo - запись в файл
cat - просотр содержимого файла
rm - удалить файл
git init - создает пустой репозиторий
git status - текущее состояние репозитория
git add имя файла - добавление файла в коммит
git commit -am 'сообщение' - добавление изменений в индекс с комментарием

nano - редактировать содержимое файла
clear - очистить консоль

git push - загрузка в удаленный репозиторий
